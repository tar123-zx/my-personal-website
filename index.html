    <script>

        document.addEventListener('DOMContentLoaded', function() {

            const mainContent = document.getElementById('main-content');

            const navLinks = {

                home: document.getElementById('nav-home'),

                portfolio: document.getElementById('nav-portfolio'),

                trivia: document.getElementById('nav-trivia')

            };

            const templates = {

                home: document.getElementById('template-home').innerHTML,

                portfolio: document.getElementById('template-portfolio').innerHTML,

                trivia: document.getElementById('template-trivia').innerHTML

            };


            // --- Page Navigation Logic ---

            function showPage(pageName) {

                // Update main content

                mainContent.innerHTML = templates[pageName];


                // Update active link style

                for (const link in navLinks) {

                    navLinks[link].classList.remove('active');

                }

                navLinks[pageName].classList.add('active');


                // If the trivia page is loaded, attach its event listeners

                if (pageName === 'trivia') {

                    attachTriviaListeners();

                }

            }

            

            // --- Event Listeners for Navigation ---

            navLinks.home.addEventListener('click', (e) => {

                e.preventDefault();

                showPage('home');

                window.location.hash = 'home';

            });

            navLinks.portfolio.addEventListener('click', (e) => {

                e.preventDefault();

                showPage('portfolio');

                window.location.hash = 'portfolio';

            });

            navLinks.trivia.addEventListener('click', (e) => {

                e.preventDefault();

                showPage('trivia');

                window.location.hash = 'trivia';

            });


            // --- Trivia Page (BMI Calculator) Logic ---

            function attachTriviaListeners() {

                const calculateBtn = document.getElementById('calculate-bmi-btn');

                const heightInput = document.getElementById('height');

                const weightInput = document.getElementById('weight');

                const resultDiv = document.getElementById('bmi-result');


                calculateBtn.addEventListener('click', function() {

                    const height = parseFloat(heightInput.value);

                    const weight = parseFloat(weightInput.value);


                    if (isNaN(height) || isNaN(weight) || height <= 0 || weight <= 0) {

                        resultDiv.innerHTML = 'Please enter valid height and weight.';

                        resultDiv.style.backgroundColor = '#fed7d7'; // Light Red

                        resultDiv.style.color = '#c53030'; // Dark Red

                        resultDiv.classList.remove('bmi-result-hidden');

                        return;

                    }


                    const heightInMeters = height / 100;

                    const bmi = weight / (heightInMeters * heightInMeters);

                    const bmiRounded = bmi.toFixed(1);


                    let category = '';

                    let color = '';

                    let bgColor = '';


                    if (bmi < 18.5) {

                        category = 'Underweight';

                        bgColor = '#cceeff'; // Light Blue

                        color = '#0077cc';

                    } else if (bmi >= 18.5 && bmi < 25) {

                        category = 'Normal weight';

                        bgColor = '#d4edda'; // Light Green

                        color = '#155724';

                    } else if (bmi >= 25 && bmi < 30) {

                        category = 'Overweight';

                        bgColor = '#fff3cd'; // Light Yellow

                        color = '#856404';

                    } else {

                        category = 'Obese';

                        bgColor = '#f8d7da'; // Light Red

                        color = '#721c24';

                    }


                    resultDiv.innerHTML = `Your BMI is ${bmiRounded}.<br>Category: ${category}`;

                    resultDiv.style.backgroundColor = bgColor;

                    resultDiv.style.color = color;

                    resultDiv.classList.remove('bmi-result-hidden');

                });

            }


            // --- Contact Form Logic ---

            const contactForm = document.getElementById('contact-form');

            contactForm.addEventListener('submit', function(e) {

                e.preventDefault();

                alert('Thank you for your message!');

                contactForm.reset();

            });

            

            // --- Initial Page Load Logic ---

            const initialPage = window.location.hash.replace('#', '') || 'home';

            if (templates[initialPage]) {

                showPage(initialPage);

            } else {

                showPage('home');

            }

        });

    </script>
